91 logger.debug("Setting up startup hooks")

93 @hook.subscribe.startup_once

96 logger.info("Qtile startup_once hook called")

98 @hook.subscribe.startup_once

101 self.run_autostart_script()

103 @hook.subscribe.startup_complete
104
105 def setup_color_watching(): # Used by qtile hook system
106 logger.info("Qtile startup completed - starting color monitoring")
107
108 # Initialize notification system
109 try:
110 show_popup_notification(
111 "Qtile Startup", "Notification system initialized", "normal"
112 )
113 logger.info("✅ Notification system initialized successfully")
114 except Exception as e:
115 logger.warning(f"❌ Failed to initialize notification system: {e}")
116
117 try:
118 # Check if already monitoring
119 if self.color_manager and self.color_manager.is_monitoring():
120 logger.info("Color monitoring already active")
121 return
122
123 # Attempt to start monitoring
124 logger.info("Starting color file monitoring...")
125 if self.color_manager:
126 self.color_manager.start_monitoring()
127
128 # Verify it started successfully
129 if self.color_manager.is_monitoring():
130 logger.info("✅ Color monitoring started successfully")
131 logger.info(f"Watching: {self.color_manager.colors_file}")
132 else:
133 logger.warning("❌ Color monitoring failed to start")
134 # Try force start as fallback
135 logger.info("Attempting force start...")
136 if hasattr(self.color_manager, "force_start_monitoring"):
137 result = self.color_manager.force_start_monitoring()
138 logger.info(f"Force start result: {result}")
139
140 except Exception as e:
141 logger.error(f"Failed to start color monitoring: {e}")
142 logger.error("Color changes won't trigger automatic qtile restart")
143 logger.error("Use Super + Ctrl + C for manual color reload")

145 @hook.subscribe.startup_complete

161 self, operation: str, func: Callable[..., Any], *args: Any, **kwargs: Any
162 ) -> bool:
163 try:
164 func(*args, **kwargs)
165 logger.info(f"✅ {operation} completed successfully")
166 return True
167 except Exception as e:
168 logger.error(f"❌ {operation} failed: {e}")
169 if operation.startswith(("Color monitoring", "Screen")):
170 logger.error(traceback.format_exc())

184 def _validate_setting(
185 self,
186 setting_name: str,
187 value: Any,
188 min_val: float = 0,
189 max_val: float | None = None,
190 warning_threshold: float | None = None,
191 ) -> tuple[bool, str | None]:
192 if not isinstance(value, int | float) or value < min_val:
193 return False, f"{setting_name} must be a number ≥ {min_val}"
194
195 if max_val and value > max_val:
196 return False, f"{setting_name} is too high (>{max_val})"
197
198 if warning_threshold and value > warning_threshold:
199 logger.warning(f"{setting_name} is quite high (>{warning_threshold})")
200
201 return True, None

209 def _handle_screen_change_event(self, event: Any = None) -> None:
210 time.sleep(self.config.screen_settings["detection_delay"])
211
212 startup_time = (
213 getattr(self.color_manager, "_startup_time", time.time())
214 if self.color_manager
215 else time.time()
216 )
217 current_time = time.time()
218
219 if current_time - startup_time > self.config.screen_settings["startup_delay"]:
220 logger.info("Screen change detected - checking for monitor changes")
221 if event is not None:
222 logger.debug(f"Screen change event data: {event}")
223
224 if self._refresh_and_check_screens():
225 logger.info("Monitor configuration changed - updating screens")
226 self._reconfigure_screens()
227 else:
228 logger.info("Screen change detected but count unchanged")
229 else:
230 logger.info("Screen change detected but ignored (too soon after startup)")

375 def validate_screen_configuration(self) -> dict[str, Any]:
376 validation: dict[str, Any] = {"valid": True, "warnings": [], "errors": []}
377
378 try:
379 delay = self.config.screen_settings["detection_delay"]
380 is_valid, error = self._validate_setting(
381 "Detection delay", delay, 0, None, 5
382 )
383 if not is_valid:
384 validation["errors"].append(error)
385 validation["valid"] = False
386
387 startup_delay = self.config.screen_settings["startup_delay"]
388 is_valid, error = self._validate_setting(
389 "Startup delay", startup_delay, 0, None, 60
390 )
391 if not is_valid:
392 validation["errors"].append(error)
393 validation["valid"] = False
394
395 except (AttributeError, KeyError, TypeError) as e:
396 validation["errors"].append(f"Missing or invalid screen settings: {e}")
397 validation["valid"] = False
398
399 return validation
