142 def validate_dpi_config(self) -> dict[str, Any]:
143 result = self._create_validation_result()
144
145 dpi_settings = self._check_config_attribute("dpi_settings", result)
146 if not dpi_settings:
147 return result
148
149 # Validate DPI value
150 dpi = dpi_settings.get("dpi", 96)
151 self._validate_numeric_range(dpi, 1, 1000, "DPI value", result, (72, 300))
152
153 # Validate scaling factor
154 scale = dpi_settings.get("scale_factor", 1.0)
155 self._validate_numeric_range(scale, 0.1, 10, "scale factor", result, (0.5, 3.0))
156
157 # Validate auto-detection setting
158 auto_detect = dpi_settings.get("auto_detect", True)
159 if not isinstance(auto_detect, bool):
160 result["warnings"].append("Auto-detect should be boolean")
161
162 return result

168 def validate_font_config(self) -> dict[str, Any]:
169 result = self._create_validation_result()
170
171 font_settings = self._check_config_attribute("font_settings", result)
172 if not font_settings:
173 return result
174
175 # Validate main font
176 main_font = font_settings.get("font", "Monospace")
177 if not isinstance(main_font, str) or not main_font.strip():
178 result["errors"].append("Invalid main font specification")
179 result["valid"] = False
180
181 # Validate font size
182 font_size = font_settings.get("fontsize", 12)
183 self._validate_numeric_range(font_size, 1, 100, "font size", result, (8, 24))
184
185 # Check font availability
186 if self._is_command_available("fc-list") and not self._check_font_available(
187 main_font
188 ):
189 result["warnings"].append(f"Font '{main_font}' may not be available")
190

239 def validate_hotkey_config(self) -> dict[str, Any]:
240 result = self._create_validation_result()
241
242 if not hasattr(self.config, "keys"):
243 result["errors"].append("No key bindings configured")
244 result["valid"] = False
245 return result
246
247 keys = self.config.keys
248 if not keys:
249 result["warnings"].append("No key bindings found")
250 return result
251
252 # Validate key structure
253 for i, key in enumerate(keys):
254 if not hasattr(key, "key") or not hasattr(key, "modifiers"):
255 result["errors"].append(f"Key {i} missing required attributes")
256 result["valid"] = False
257 continue
258
259 if not key.key or not key.key.strip():
260 result["warnings"].append(f"Key {i} has empty key value")
261
262 if not isinstance(key.modifiers, list):
263 result["warnings"].append(f"Key {i} modifiers should be a list")
264
265 return result

332 def get_validation_summary(self) -> str:
333 results = self.validation_results
334 lines = [
335 "=== Qtile Configuration Validation Summary ===",
336 f"Overall Status: {' Valid' if results['valid'] else ' Invalid'}",
337 "",
338 ]
339
340 if results["errors"]:
341 lines.append(f"Errors ({len(results['errors'])}):")
342 lines.extend(f" {error}" for error in results["errors"])
343
344 if results["warnings"]:
345 lines.append(f"Warnings ({len(results['warnings'])}):")
346 lines.extend(f" {warning}" for warning in results["warnings"])
347
348 if not results["errors"] and not results["warnings"]:
349 lines.append(" No issues found!")
350
351 return "\n".join(lines)
