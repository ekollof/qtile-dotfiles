704 def apply_bsd_customizations(self) -> None:
705 if not self.platform_info.is_bsd:
706 return
707
708 # Get BSD-specific overrides
709 bsd_config = self.get_bsd_specific_overrides()
710
711 # Apply overrides to platform config
712 for key, value in bsd_config.items():
713 self.platform_config.add_override(key, value)
714
715 # BSD-specific application overrides (examples)
716 if self.platform_info.is_openbsd:
717 # OpenBSD often has different default applications
718 self.platform_config.add_override("text_editor", "vi")
719 self.platform_config.add_override("pdf_viewer", "xpdf")
720 self.platform_config.add_override("image_viewer", "xv")
721
722 # BSD systems might need different font settings
723 if self.platform_info.is_bsd:
724 # Fallback to more common fonts on BSD
725 self.platform_config.add_override("fallback_font", "DejaVu Sans Mono")
726
727
728# Create global config instance
729config = QtileConfig()
730
731# Apply BSD-specific customizations if running on BSD
732if config.platform_info.is_bsd:
733 config.apply_bsd_customizations()
