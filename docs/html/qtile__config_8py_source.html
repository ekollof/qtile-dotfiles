567 def apply_bsd_customizations(self) -> None:
568 if not self.platform_info.is_bsd:
569 return
570
571 # Get BSD-specific overrides
572 bsd_config = self.get_bsd_specific_overrides()
573
574 # Apply overrides to platform config
575 for key, value in bsd_config.items():
576 self.platform_config.add_override(key, value)
577
578 # BSD-specific application overrides (examples)
579 if self.platform_info.is_openbsd:
580 # OpenBSD often has different default applications
581 self.platform_config.add_override("text_editor", "vi")
582 self.platform_config.add_override("pdf_viewer", "xpdf")
583 self.platform_config.add_override("image_viewer", "xv")
584
585 # BSD systems might need different font settings
586 if self.platform_info.is_bsd:
587 # Fallback to more common fonts on BSD
588 self.platform_config.add_override("fallback_font", "DejaVu Sans Mono")
589
590
591# Create global config instance
592config = QtileConfig()
593
594# Apply BSD-specific customizations if running on BSD
595if config.platform_info.is_bsd:
596 config.apply_bsd_customizations()
