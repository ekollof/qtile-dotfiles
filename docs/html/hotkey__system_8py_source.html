36 @staticmethod

38 def format_key_combination(key_combo: str) -> str:
39 # Replace modifier names with more readable versions
40 formatted = key_combo.replace("mod4", "Super")
41 formatted = formatted.replace("mod1", "Alt")
42 formatted = formatted.replace("shift", "Shift")
43 formatted = formatted.replace("control", "Ctrl")
44
45 # Capitalize single keys
46 parts = formatted.split("+")
47 if len(parts) > 1:
48 # Last part is the main key
49 parts[-1] = parts[-1].capitalize()
50 else:
51 parts[0] = parts[0].capitalize()
52
53 return "+".join(parts)
54
55 @staticmethod
56
57 def extract_key_combination(key: Any) -> str:
58 # Extract modifiers and key
59 modifiers = key.modifiers if key.modifiers else []
60 key_name = key.key
61
62 # Combine modifiers and key
63 key_combo = "+".join(modifiers) + "+" + key_name if modifiers else key_name
64
65 return key_combo

67 @staticmethod
68
69 def infer_description(key: Any) -> str:
70 # Try to get explicit description first
71 description = getattr(key, "desc", None)
72 if description:
73 return description
74
75 # Try to infer from commands
76 if hasattr(key, "commands") and key.commands:
77 cmd = key.commands[0]
78
79 if hasattr(cmd, "__name__"):
80 return cmd.__name__.replace("_", " ").title()
81 elif hasattr(cmd, "name"):
82 return cmd.name.replace("_", " ").title()
83 else:
84 return KeyFormatter._parse_command_string(cmd)
85
86 return "Custom action"

88 @staticmethod
89
92 def _parse_command_string(cmd: Any) -> str:
93 cmd_str = str(cmd).lower()
94
95 # Use match statement for cleaner pattern matching
96 match True:
97 case _ if "spawn" in cmd_str:
98 # Extract spawn command
99 if hasattr(cmd, "args") and cmd.args:
100 app_name = (
101 cmd.args[0].split("/")[-1]
102 if "/" in cmd.args[0]
103 else cmd.args[0]
104 )
105 return f"Launch {app_name}"
106 else:
107 return "Launch application"
108 case _ if "layout" in cmd_str:
109 return "Change layout"
110 case _ if "group" in cmd_str:
111 return "Switch group"
112 case _ if "window" in cmd_str:
113 return "Window action"
114 case _:
115 return str(cmd)[:50] # Truncate long descriptions

153 def categorize_key(self, key: Any) -> str:
154 key_name = key.key
155 description = KeyFormatter.infer_description(key).lower()
156
157 # Check if it's a number key (likely group/workspace)
158 if key_name.isdigit():
159 return "Group/Workspace"
160
161 # Define keyword categories
162 window_words = ["window", "focus", "move", "close", "kill", "floating"]
163 layout_words = ["layout", "tile", "max", "split"]
164 group_words = ["group", "workspace"]
165 system_words = [
166 "restart",
167 "quit",
168 "shutdown",
169 "reload",
170 "screen",
171 "color",
172 ]
173 app_words = ["launch", "spawn", "browser", "terminal"]
174 display_words = ["screen", "monitor", "display"]
175
176 # Categorize based on description content using match statements
177 match True:
178 case _ if any(word in description for word in window_words):
179 return "Window Management"
180 case _ if any(word in description for word in layout_words):
181 return "Layout Control"
182 case _ if any(word in description for word in group_words):
183 return "Group/Workspace"
184 case _ if any(word in description for word in system_words):
185 return "System"
186 case _ if any(word in description for word in app_words):
187 return "Applications"
188 case _ if any(word in description for word in display_words):
189 return "Screen/Display"
190 case _:
191 return "Other"

194 def add_key_to_category(self, key: Any, category: str | None = None):
195 if category is None:
196 category = self.categorize_key(key)
197
198 # Extract key combination and description
199 key_combo = KeyFormatter.extract_key_combination(key)
200 description = KeyFormatter.infer_description(key)
201
202 # Format the hotkey line
203 hotkey_line = KeyFormatter.format_hotkey_line(key_combo, description)
204
205 # Add to category
206 if category in self.categories:
207 self.categories[category].append(hotkey_line)
208 else:
209 self.categories["Other"].append(hotkey_line)

212 def process_keys(self, keys: list[Any]) -> dict[str, list[str]]:
213 self.clear_categories()
214
215 for key in keys:
216 self.add_key_to_category(key)
217
218 # Sort hotkeys within each category
219 for category_hotkeys in self.categories.values():
220 category_hotkeys.sort()
221
222 return self.categories

225 def build_formatted_list(self, include_instructions: bool = True) -> list[str]:
226 final_hotkeys = []
227
228 # Add instructions if requested
229 if include_instructions:
230 final_hotkeys.extend(KeyFormatter.create_instructions())
231
232 # Add categories with hotkeys
233 for category_name, category_hotkeys in self.categories.items():
234 if category_hotkeys: # Only show categories that have hotkeys
235 final_hotkeys.append("") # Empty line for separation
236 final_hotkeys.append(f"=== {category_name} ===")
237 final_hotkeys.extend(category_hotkeys)
238
239 return final_hotkeys

458 def show_hotkeys(self):
459 try:
460 # Get hotkey list
461 hotkeys = self._get_hotkey_list()
462 rofi_input = "\n".join(hotkeys)
463
464 # Create temporary theme file
465 rofi_theme = self.theme_manager.get_rofi_theme()
466
467 with tempfile.NamedTemporaryFile(
468 mode="w", suffix=".rasi", delete=False
469 ) as theme_file:
470 theme_file.write(rofi_theme)
471 theme_file_path = theme_file.name
472
473 try:
474 # Get rofi command
475 cmd = self.theme_manager.get_rofi_command_args(theme_file_path)
476
477 logger.info("Showing hotkey display")
478
479 # Run rofi
480 subprocess.run(
481 cmd,
482 input=rofi_input,
483 text=True,
484 capture_output=True,
485 timeout=30,
486 )
487
488 logger.debug("Hotkey display closed")
489
490 finally:
491 # Clean up temporary theme file
492 with contextlib.suppress(Exception):
493 os.unlink(theme_file_path)
494
495 except subprocess.TimeoutExpired:
496 logger.warning("Hotkey display timed out")
497 except FileNotFoundError:
498 logger.error("rofi not found - please install rofi to use hotkey display")
499 self._show_fallback_notification()
500 except Exception as e:
501 logger.error(f"Error showing hotkeys: {e}")
