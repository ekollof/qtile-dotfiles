172 def build(self, viewbox: str | None = None) -> str:
173 if viewbox is None:
174 viewbox = f"0 0 {self.width} {self.height}"
175
176 svg = [
177 f'<svg width="{self.width}" height="{self.height}" '
178 f'viewBox="{viewbox}" xmlns="http://www.w3.org/2000/svg">',
179 ]
180
181 if self.defs:
182 svg.append("<defs>")
183 svg.extend(self.defs)
184 svg.append("</defs>")
185
186 svg.extend(self.elements)
187 svg.append("</svg>")
188
189 return "\n".join(svg)

212 def load_svg(self, file_path: Path | str) -> SVGIcon | None:

252 def recolor_svg(self, svg_icon: SVGIcon, color_map: dict[str, str]) -> SVGIcon:
253 content = svg_icon.content
254
255 for old_color, new_color in color_map.items():
256 # Normalize color format
257 old_color = old_color.lower().strip()
258 old_hex = old_color[1:] if old_color.startswith("#") else old_color
259
260 # Replace various color formats
261 patterns = [
262 rf'fill="#{old_hex}"',
263 rf'stroke="#{old_hex}"',
264 rf'fill="{old_color}"',
265 rf'stroke="{old_color}"',
266 rf"fill:\s*#{old_hex}",
267 rf"stroke:\s*#{old_hex}",
268 rf'stop-color="#{old_hex}"',
269 rf'stop-color="{old_color}"',
270 ]
271
272 replacements = [
273 f'fill="{new_color}"',
274 f'stroke="{new_color}"',
275 f'fill="{new_color}"',
276 f'stroke="{new_color}"',
277 f"fill:{new_color}",
278 f"stroke:{new_color}",
279 f'stop-color="{new_color}"',
280 f'stop-color="{new_color}"',
281 ]
282
283 for pattern, replacement in zip(patterns, replacements, strict=False):
284 content = re.sub(pattern, replacement, content, flags=re.IGNORECASE)
285
286 return SVGIcon(
287 content=content,
288 width=svg_icon.width,
289 height=svg_icon.height,
290 viewbox=svg_icon.viewbox,
291 )

478 builder = SVGBuilder(self.size, self.size)
479
480 if not connected:
481 # Disconnected WiFi with X
482 builder.add_path(
483 "M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2zM8.5 7.5l7 7-1 1-7-7 1-1z",
484 fill=self.colors["muted"],
485 )
486 return builder.build()
487
488 # WiFi signal arcs
489 center_x, center_y = self.size // 2, self.size // 2 + 2
490
491 # Signal strength arcs (from innermost to outermost)
492 arcs = [
493 (3, 1.5), # Innermost arc
494 (6, 2.5), # Middle arc
495 (9, 3.5), # Outermost arc
496 ]
497
498 for i, (radius, stroke_width) in enumerate(arcs):
499 color = self.colors["accent"] if strength > i else self.colors["muted"]
500
501 # Create arc using path
502 x1 = center_x + radius * 0.707 # cos(45)
503 y1 = center_y + radius * 0.707 # sin(45)
504 x2 = center_x - radius * 0.707
505 y2 = center_y + radius * 0.707
506
507 arc_path = f"M {x1} {y1} A {radius} {radius} 0 0 0 {x2} {y2}"
508 builder.add_path(
509 arc_path, fill="none", stroke=color, stroke_width=stroke_width
510 )

512 # Center dot
513 builder.add_circle(center_x, center_y, 1.5, fill=self.colors["foreground"])
