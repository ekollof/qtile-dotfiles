64 def smart_maximize(qtile: Any) -> None:
65 try:
66 current_window = qtile.current_window
67 if not current_window:
68 logger.debug("No current window to maximize")
69 return
70
71 # Check if window is currently maximized
72 is_maximized = getattr(current_window, "maximized", False)
73 current_group = qtile.current_group
74
75 logger.debug(f"Smart maximize called for window: {current_window.name}")
76 logger.debug(f"Window currently maximized: {is_maximized}")
77 logger.debug(f"Current group: {current_group.name}")
78 logger.debug(f"Windows in group: {[w.name for w in current_group.windows]}")
79
80 if is_maximized:
81 # Un-maximize: restore the window and bring back other windows
82 logger.info("Un-maximizing window and restoring other windows")
83 current_window.toggle_maximize()
84
85 # Restore minimized windows in current group
86 restored_count = 0
87 for window in current_group.windows:
88 if (
89 window != current_window
90 and getattr(window, "minimized", False)
91 and getattr(window, "_smart_maximized_hidden", False)
92 ):
93 window.toggle_minimize()
94 # Clear the flag
95 window._smart_maximized_hidden = False
96 restored_count += 1
97 logger.debug(f"Restored window: {window.name}")
98
99 logger.info(f"Restored {restored_count} windows")
100
101 else:
102 # Maximize: hide other windows and maximize current
103 logger.info("Maximizing window and hiding others")
104
105 # First, minimize all other windows in the current group
106 hidden_count = 0
107 for window in current_group.windows:
108 if (
109 window != current_window
110 and not getattr(window, "minimized", False)
111 and not window.floating
112 ): # Don't minimize floating windows
113 # Mark that this window was hidden by smart maximize
114 window._smart_maximized_hidden = True
115 window.toggle_minimize()
116 hidden_count += 1
117 logger.debug(f"Minimized window: {window.name}")
118
119 logger.info(f"Minimized {hidden_count} windows")
120
121 # Then maximize the current window
122 current_window.toggle_maximize()
123 logger.info(f"Maximized window: {current_window.name}")
124
125 except Exception as e:
126 logger.error(f"Error in smart maximize: {e}")
127 # Fallback to regular maximize if something goes wrong
128 try:
129 qtile.current_window.toggle_maximize()
130 logger.info("Fallback: Used regular maximize")
131 except Exception as fallback_error:
132 logger.error(f"Fallback maximize also failed: {fallback_error}")

137 def _warp_mouse_to_window(self, qtile: Any) -> None:
138 try:
139 # Check if mouse warping is enabled
140 if not self.qtile_config.mouse_warp_focus:
141 return
142
143 current_window = qtile.current_window
144 if not current_window:
145 return
146
147 # Get window geometry
148 x = current_window.x
149 y = current_window.y
150 width = current_window.width
151 height = current_window.height
152
153 # Calculate center position
154 center_x = x + (width // 2)
155 center_y = y + (height // 2)
156
157 # Warp mouse to center of window
158 qtile.core.warp_pointer(center_x, center_y)
159 logger.debug(f"Warped mouse to window center: ({center_x}, {center_y})")
160
161 except Exception as e:
162 logger.debug(f"Error warping mouse to window: {e}")

568 layout_name = qtile.current_group.layout.name.lower()
569
570 try:
571 match layout_name:
572 case "monadtall" | "monadwide":
573 # MonadTall/Wide: grow main window
574 qtile.current_group.layout.grow()
575 case "tile":
576 # Tile: increase ratio
577 qtile.current_group.layout.increase_ratio()
578 case "bsp":

580 qtile.current_group.layout.grow_right()
581 case "matrix":
582 # Matrix: add column (horizontal growth)

584 case _:
585 # Max and Floating layouts: no-op (but don't error)
586 pass
