37 def get_available_font(
38 self,
39 preferred_font: str | None = None,
40 fallback_fonts: list[str] | None = None
41 ) -> str:
42 # Default fallback fonts (common monospace fonts across platforms)
43 if fallback_fonts is None:
44 fallback_fonts = [
45 "JetBrains Mono",
46 "Fira Code",
47 "Source Code Pro",
48 "Hack",
49 "Inconsolata",
50 "DejaVu Sans Mono",
51 "Liberation Mono",
52 "Consolas", # Windows
53 "Monaco", # macOS
54 "Monospace", # Generic Linux
55 "monospace" # Lowercase fallback
56 ]
57
58 # Build preference list: user preference first, then fallbacks
59 fonts_to_try = []
60 if preferred_font:
61 fonts_to_try.append(preferred_font)
62 fonts_to_try.extend(fallback_fonts)
63
64 # Check cache first
65 cache_key = (preferred_font, tuple(fallback_fonts))
66 if cache_key in self._font_cache:
67 return self._font_cache[cache_key]
68
69 # Try each font in order
70 for font in fonts_to_try:
71 if self._is_font_available(font):
72 logger.debug(f"Selected font: {font}")
73 self._font_cache[cache_key] = font
74 return font
75
76 # Ultimate fallback
77 final_fallback = "monospace"
78 logger.warning(
79 f"No fonts available from preferences, using final fallback: {final_fallback}"
80 )
81 self._font_cache[cache_key] = final_fallback
82 return final_fallback

91 def _is_font_available(self, font_name: str) -> bool:
92 try:
93 match self.system:
94 case "linux":
95 return self._check_font_linux(font_name)
96 case "openbsd" | "freebsd" | "netbsd" | "dragonfly":
97 return self._check_font_bsd(font_name)
98 case "darwin":
99 return self._check_font_macos(font_name)
100 case _:
101 # Unknown system: assume basic fonts are available
102 return font_name.lower() in ["monospace", "mono", "sans-serif", "serif"]
103 except Exception as e:
104 logger.debug(f"Error checking font {font_name}: {e}")
105 return False

115 def _check_font_linux(self, font_name: str) -> bool:
116 try:
117 # Use fc-match to check if font resolves to itself (preferred method)
118 result = subprocess.run(
119 ['fc-match', '--format=%{family}', font_name],
120 capture_output=True,
121 text=True,
122 timeout=5
123 )
124 if result.returncode == 0:
125 matched_family = result.stdout.strip()
126 # If fc-match returns the same font name, it's available
127 return font_name.lower() == matched_family.lower()
128 except (subprocess.TimeoutExpired, FileNotFoundError) as e:
129 logger.debug(f"fc-match failed for {font_name}: {e}")
130
131 # Backup method: try fc-list as fallback
132 try:
133 result = subprocess.run(
134 ['fc-list', ':', 'family'],
135 capture_output=True,
136 text=True,
137 timeout=5
138 )
139 if result.returncode == 0:
140 return font_name.lower() in result.stdout.lower()
141 except (subprocess.TimeoutExpired, FileNotFoundError) as e:
142 logger.debug(f"fc-list failed for {font_name}: {e}")
143
144 # Final fallback: assume basic fonts are available
145 return font_name.lower() in ["monospace", "mono"]

154 def _check_font_bsd(self, font_name: str) -> bool:
155 try:
156 # Try fc-match if available (on systems with fontconfig)
157 result = subprocess.run(
158 ['fc-match', '--format=%{family}', font_name],
159 capture_output=True,
160 text=True,
161 timeout=5
162 )
163 if result.returncode == 0:
164 matched_family = result.stdout.strip()
165 return font_name.lower() == matched_family.lower()
166 except (subprocess.TimeoutExpired, FileNotFoundError):
167 pass
168
169 # Fallback: check common font directories
170 font_dirs = [
171 Path('/usr/X11R6/lib/X11/fonts'),
172 Path('/usr/local/share/fonts'),
173 Path('/usr/share/fonts'),
174 Path('~/.fonts').expanduser(),
175 ]
176
177 for font_dir in font_dirs:
178 if font_dir.exists():
179 try:
180 for font_file in font_dir.rglob('*'):
181 if (font_file.is_file() and
182 font_file.suffix.lower() in ('.ttf', '.otf', '.pfb', '.pcf') and
183 font_name.replace(' ', '').lower() in font_file.name.lower()):
184 return True
185 except Exception:
186 continue
187
188 # Default fallback fonts that should be available
189 return font_name.lower() in ["monospace", "mono", "fixed"]
