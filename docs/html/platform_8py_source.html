330 def get_application(self, app_type: str, fallback: str | None = None) -> str:
331 system = self.platform.system
332
333 if app_type in self._application_preferences:
334 # Use match statement for cleaner platform-specific logic
335 match system:
336 case "openbsd" | "freebsd" | "netbsd" | "dragonfly":
337 preferences = self._application_preferences[app_type].get(
338 system,
339 self._application_preferences[app_type].get("openbsd", []),
340 )
341 case "linux":
342 preferences = self._application_preferences[app_type]["linux"]
343 case _:
344 preferences = self._application_preferences[app_type].get(
345 system,
346 self._application_preferences[app_type].get("linux", []),
347 )
348
349 app = self.platform.get_preferred_application(app_type, preferences)
350 if app:
351 return app
352
353 # If no preferred app found, return fallback or default
354 return fallback or "xterm"

362 def get_command(self, command_type: str, fallback: str | None = None) -> str:
363 system = self.platform.system
364
365 # Use match statement for platform-specific command selection
366 match system:
367 case "linux":
368 command = self._config_overrides.get("linux", {}).get(command_type)
369 case "openbsd":
370 command = self._config_overrides.get("openbsd", {}).get(command_type)
371 case "freebsd":
372 command = self._config_overrides.get("freebsd", {}).get(command_type)
373 case "netbsd":
374 command = self._config_overrides.get("netbsd", {}).get(command_type)
375 case _:
376 command = self._config_overrides.get(system, {}).get(command_type)
377
378 if command:
379 # Verify the base command exists
380 base_cmd = command.split()[0]
381 if self.platform.has_command(base_cmd):
382 return command
383
384 # Return fallback or try Linux defaults
385 if fallback:
386 return fallback
387
388 linux_defaults = self._config_overrides.get("linux", {})
389 return linux_defaults.get(command_type, "")
