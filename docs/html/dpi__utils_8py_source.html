121 def _parse_xrandr_line(self, line: str) -> float | None:
122 parts = line.split()
123 resolution_part = None
124 mm_part = None
125
126 # Find resolution part (e.g., "1920x1080+0+0" -> "1920x1080")
127 for i, part in enumerate(parts):
128 if (
129 "x" in part
130 and part.replace("x", "").replace("+", "").replace("-", "").isdigit()
131 ):
132 resolution_part = part.split("+")[0] # Get resolution before position
133 # Find physical dimensions (e.g., "597mm x 336mm")
134 if part.endswith("mm"):
135 mm_part = parts[i - 2 : i + 1] # Get "597mm x 336mm"
136 break
137
138 if resolution_part and mm_part and len(mm_part) >= 3:
139 try:
140 # Extract pixel width and physical width in millimeters
141 width_px = int(resolution_part.split("x")[0])
142 width_mm = float(mm_part[0].replace("mm", ""))
143 # Calculate DPI: pixels per inch = pixels / (mm / 25.4 mm/inch)
144 return round(width_px / (width_mm / 25.4))
145 except (ValueError, IndexError):
146 pass
147 return None
