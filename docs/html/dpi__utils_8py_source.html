63 def _try_xdpyinfo(self) -> float | None:
64 try:
65 result = subprocess.run(['xdpyinfo'], capture_output=True, text=True, timeout=2)
66 if result.returncode == 0:
67 for line in result.stdout.split('\n'):
68 if 'dots per inch' in line.lower():
69 # Parse: "resolution: 96x96 dots per inch"

72 logger.info(f"Detected DPI via xdpyinfo: {x_dpi}")

74 except (subprocess.TimeoutExpired, FileNotFoundError, ValueError, IndexError):

106 def _parse_xrandr_line(self, line: str) -> float | None:
107 parts = line.split()
108 resolution_part = None
109 mm_part = None
110
111 # Find resolution part (e.g., "1920x1080+0+0" -> "1920x1080")
112 for i, part in enumerate(parts):
113 if 'x' in part and part.replace('x', '').replace('+', '').replace('-', '').isdigit():
114 resolution_part = part.split('+')[0] # Get resolution before position
115 # Find physical dimensions (e.g., "597mm x 336mm")
116 if part.endswith('mm'):
117 mm_part = parts[i-2:i+1] # Get "597mm x 336mm"
118 break
119
120 if resolution_part and mm_part and len(mm_part) >= 3:
121 try:
122 # Extract pixel width and physical width in millimeters
123 width_px = int(resolution_part.split('x')[0])
124 width_mm = float(mm_part[0].replace('mm', ''))
125 # Calculate DPI: pixels per inch = pixels / (mm / 25.4 mm/inch)
126 return round(width_px / (width_mm / 25.4))
127 except (ValueError, IndexError):
128 pass
129 return None
